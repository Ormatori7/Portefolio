document.addEventListener("DOMContentLoaded", function () {
        const phoneLink = document.getElementById("click");
        let isFond2 = true; // Variable pour alterner les fonds
        let isApple = true; // Variable pour alterner les images

        phoneLink.addEventListener("click", function (event) {
            event.preventDefault(); // Empêche le rechargement de la page

            const phone = phoneLink.querySelector("img"); // Sélectionne l'image

            if (phone.style.transform === "scale(2.3) rotate(-90deg) translateY(-200px)") {
                phone.style.transform = "scale(1)"; // Réinitialise la taille
                phone.src = "../image/apple.png"
            } else {
                phone.style.transform = "scale(2.3) rotate(-90deg) translateY(-200px)";
                phone.src = "../image/apple2.png"
                phone.style.backfaceVisibility = ''

                setTimeout(() => {

                    // Change l'image après la transition
                    phone.src = isApple ? "../image/apple2.png" : "../image/apple.png";

                    // isApple = !isApple; // Inverse l'état
                }, 1); // Attendre la durée de la transition (1s)
            }

            phone.style.transition = "transform 1s ease"; // Animation fluide
        });
    });
